# Complete Walkthrough

This walkthrough demonstrates IntentusNet's core capabilities with a realistic scenario: a search service with fallback routing.

## Scenario Overview

**Intent**: "Search for information with automatic fallback"

**Requirements**:
1. Primary search agent attempts first
2. If primary fails, fallback to secondary agent
3. Router decisions are deterministic and recorded
4. Execution can be replayed without re-running agents

## Part 1: Basic Agent Setup

Create `search_demo.py`:

```python
import json
from intentusnet import (
    IntentusRuntime,
    BaseAgent,
    IntentEnvelope,
    AgentResponse,
    AgentDefinition,
    Capability,
    IntentRef,
    RoutingStrategy,
)

# =============================================================================
# AGENT DEFINITIONS
# =============================================================================

class PrimarySearchAgent(BaseAgent):
    """Primary search agent - may fail on first attempt."""

    def __init__(self, router):
        definition = AgentDefinition(
            name="search-primary",
            nodePriority=10,  # Lower = higher priority
            capabilities=[
                Capability(intent=IntentRef(name="SearchIntent", version="1.0"))
            ],
        )
        super().__init__(definition=definition, router=router)
        self._fail_once = True  # Simulate failure on first call

    def handle_intent(self, env: IntentEnvelope) -> AgentResponse:
        query = env.payload.get("query", "")

        if self._fail_once:
            self._fail_once = False
            print(f"  [Primary] Failing on query: '{query}' (simulated)")
            return AgentResponse.failure(
                self.error("Primary search unavailable"),
                agent=self.definition.name,
                trace_id=env.metadata.traceId,
            )

        print(f"  [Primary] Searching for: '{query}'")
        return AgentResponse.success(
            payload={"source": "primary", "results": [f"result-1:{query}", f"result-2:{query}"]},
            agent=self.definition.name,
            trace_id=env.metadata.traceId,
        )


class FallbackSearchAgent(BaseAgent):
    """Fallback search agent - always succeeds."""

    def __init__(self, router):
        definition = AgentDefinition(
            name="search-fallback",
            nodePriority=50,  # Lower priority than primary
            capabilities=[
                Capability(intent=IntentRef(name="SearchIntent", version="1.0"))
            ],
        )
        super().__init__(definition=definition, router=router)

    def handle_intent(self, env: IntentEnvelope) -> AgentResponse:
        query = env.payload.get("query", "")
        print(f"  [Fallback] Searching for: '{query}'")
        return AgentResponse.success(
            payload={"source": "fallback", "results": [f"fallback-result:{query}"]},
            agent=self.definition.name,
            trace_id=env.metadata.traceId,
        )
```

## Part 2: Fallback Routing Demo

Add to `search_demo.py`:

```python
from intentusnet import (
    IntentContext,
    IntentMetadata,
    RoutingOptions,
    RoutingStrategy,
    Priority,
)
from intentusnet.protocol import RoutingMetadata
from intentusnet.utils.timestamps import now_iso
from intentusnet.utils.id_generator import generate_uuid

# =============================================================================
# DEMO: FALLBACK ROUTING
# =============================================================================

def create_envelope(intent_name: str, payload: dict, strategy: RoutingStrategy):
    """Helper to create an IntentEnvelope with a specific routing strategy."""
    now = now_iso()
    return IntentEnvelope(
        version="1.0",
        intent=IntentRef(name=intent_name, version="1.0"),
        payload=payload,
        context=IntentContext(
            sourceAgent="demo",
            timestamp=now,
            priority=Priority.NORMAL,
            tags=["demo"],
        ),
        metadata=IntentMetadata(
            requestId=str(generate_uuid()),
            source="demo",
            createdAt=now,
            traceId=str(generate_uuid()),
        ),
        routing=RoutingOptions(strategy=strategy),
        routingMetadata=RoutingMetadata(),
    )

def demo_fallback_routing():
    print("=" * 60)
    print("DEMO 1: Fallback Routing")
    print("=" * 60)

    # Create runtime (recording disabled to avoid serialization edge cases)
    runtime = IntentusRuntime(enable_recording=False)

    # Register agents - the runtime injects the router
    runtime.register_agent(PrimarySearchAgent)
    runtime.register_agent(FallbackSearchAgent)

    # Use FALLBACK strategy - router will try agents in priority order
    # If primary fails, it automatically falls back to next agent
    print("\n[First Request - Using FALLBACK strategy]")
    print("  Primary will fail, router will fallback to secondary")
    envelope = create_envelope(
        "SearchIntent",
        {"query": "deterministic routing"},
        RoutingStrategy.FALLBACK,
    )
    response = runtime.router.route_intent(envelope)

    print(f"\n  Status: {response.status}")
    print(f"  Agent: {response.metadata.get('agent')}")
    print(f"  Result: {json.dumps(response.payload, indent=4)}")

    print("\n[Second Request - Primary succeeds now]")
    envelope2 = create_envelope(
        "SearchIntent",
        {"query": "intent routing"},
        RoutingStrategy.FALLBACK,
    )
    response2 = runtime.router.route_intent(envelope2)

    print(f"\n  Status: {response2.status}")
    print(f"  Agent: {response2.metadata.get('agent')}")
    print(f"  Result: {json.dumps(response2.payload, indent=4)}")

    return runtime
```

**Key Point**: The `RoutingStrategy.FALLBACK` tells the router to try agents in priority order
(sorted by `nodePriority`). If one agent fails, it automatically tries the next one.

## Key Observations

| Feature | What Happened |
|---------|---------------|
| Agent Registration | Agents registered via `runtime.register_agent()` |
| Router API | `router.route_intent(envelope)` with explicit strategy |
| Fallback Routing | `RoutingStrategy.FALLBACK` tried agents in priority order |
| Deterministic Order | Agents sorted by `nodePriority` (lower = higher priority) |
| Recording | Executions saved to `.intentusnet/records/` when enabled |

## Execution Flow

```
router.route_intent(envelope)
        │
        │  envelope.routing.strategy = FALLBACK
        │
        ▼
┌───────────────────────────┐
│   IntentRouter            │  Finds agents by capability
│                           │  Sorts by nodePriority
└───────────┬───────────────┘
            │
            ▼
┌───────────────────────────┐
│   PrimarySearchAgent      │  Attempts first (priority=10)
│   (nodePriority=10)       │  → Returns error
└───────────┬───────────────┘
            │ FALLBACK (strategy)
            ▼
┌───────────────────────────┐
│   FallbackSearchAgent     │  Attempts second (priority=50)
│   (nodePriority=50)       │  → Returns success
└───────────┬───────────────┘
            │
            ▼
      AgentResponse
```
